Given an integer, the task is to find factorial of the number.
 
Input:
The first line of input contains an integer T denoting the number of test cases.  
The first line of each test case is N,the number whose factorial is to be found
 
Output:
Print the factorial of the number in separate line.
 
Constraints:
1 ≤ T ≤ 100
1 ≤ N ≤ 1000
 
Example:
Input
3
5
10
2
 
Output
120
3628800
2


import java.util.*;
import java.lang.*;
import java.io.*;
import java.math.*;

class GFG {
    static BigInteger fact(BigInteger b){
        if(b.equals(new BigInteger("0"))) return new BigInteger("1");
        return b.multiply(fact(b.subtract(new BigInteger("1"))));
    }
	public static void main (String[] args) {
		Scanner in = new Scanner(System.in);
		int t = in.nextInt();
		while(t-- > 0){
		    String s = in.next();
		    BigInteger b = new BigInteger(s);
		    System.out.println(fact(b));
		}
		in.close();
	}
}
